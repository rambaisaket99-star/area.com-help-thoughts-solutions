PYTHON
import area

# Initialize with your API key
client = area.Area(api_key="area_prod_sk_••••••••••••••••")

# Create a conversation
response = client.conversations.create(
    workspace_id="ws_production",
    channel="api",
    messages=[{
        "role": "user",
        "content": "Where is my order #12847?"
    }]
)

print(response.message)
# Output: "Your order #12847 is in transit with FedEx…"

# Stream responses
with client.conversations.stream(workspace_id="ws_production", messages=[...]) as stream:
    for chunk in stream:
        print(chunk.delta, end="", flush=True)
TYPESCRIPT
import Area from '@area-ai/sdk';

// Initialize the client
const client = new Area({
  apiKey: process.env.AREA_API_KEY,
});

// Create a conversation
const response = await client.conversations.create({
  workspaceId: 'ws_production',
  channel: 'api',
  messages: [{ role: 'user', content: 'Where is my order?' }],
});

console.log(response.message);

// Streaming
const stream = await client.conversations.stream({
  workspaceId: 'ws_production',
  messages: [{ role: 'user', content: 'Where is my order?' }],
});
for await (const chunk of stream) {
  process.stdout.write(chunk.delta ?? '');
}

GO
package main

import (
    "fmt"
    area "github.com/area-ai/area-go"
)

func main() {
    client := area.NewClient(area.WithAPIKey("area_prod_sk_••••••••"))

    resp, err := client.Conversations.Create(area.ConversationParams{
        WorkspaceID: "ws_production",
        Channel:     "api",
        Messages: []area.Message{
            {Role: "user", Content: "Where is my order #12847?"},
        },
    })
    if err != nil {
        panic(err)
    }
    fmt.Println(resp.Message)
}
